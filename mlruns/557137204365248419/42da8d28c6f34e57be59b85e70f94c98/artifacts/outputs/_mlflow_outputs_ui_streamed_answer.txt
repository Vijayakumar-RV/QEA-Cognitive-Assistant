```
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# Set up the webdriver
driver = webdriver.Chrome()

# Preconditions
# 1. The user is logged in to the application.
# 2. The user has added at least one product to the cart.
# (Assuming these preconditions are already met)

# Step 1: Click on the 'Cart' icon in the top right corner of the page.
cart_icon_locator = (By.CSS_SELECTOR, "#cart-icon")
cart_icon = WebDriverWait(driver, 10).until(EC.element_to_be_clickable(cart_icon_locator))
cart_icon.click()

# Step 2: Click on the 'Checkout' button.
checkout_button_locator = (By.CSS_SELECTOR, "#checkout-button")
checkout_button = WebDriverWait(driver, 10).until(EC.element_to_be_clickable(checkout_button_locator))
checkout_button.click()

# Step 3: Enter the shipping address details and click 'Next'.
shipping_address_locator = (By.ID, "shipping-address")
shipping_address = WebDriverWait(driver, 10).until(EC.visibility_of_element_located(shipping_address_locator))
shipping_address.send_keys("Test Shipping Address")

next_button_locator = (By.CSS_SELECTOR, "#next-button")
next_button = WebDriverWait(driver, 10).until(EC.element_to_be_clickable(next_button_locator))
next_button.click()

# Step 4: Select the payment method and click 'Next'.
payment_method_locator = (By.ID, "payment-method")
payment_method = WebDriverWait(driver, 10).until(EC.visibility_of_element_located(payment_method_locator))
payment_method.click()

next_button_locator = (By.CSS_SELECTOR, "#next-button")
next_button = WebDriverWait(driver, 10).until(EC.element_to_be_clickable(next_button_locator))
next_button.click()

# Step 5: Enter the payment details and click 'Place Order'.
payment_details_locator = (By.ID, "payment-details")
payment_details = WebDriverWait(driver, 10).until(EC.visibility_of_element_located(payment_details_locator))
payment_details.send_keys("Test Payment Details")

place_order_button_locator = (By.CSS_SELECTOR, "#place-order-button")
place_order_button = WebDriverWait(driver, 10).until(EC.element_to_be_clickable(place_order_button_locator))
place_order_button.click()

# Step 6: Verify the order summary and click 'Confirm'.
order_summary_locator = (By.ID, "order-summary")
order_summary = WebDriverWait(driver, 10).until(EC.visibility_of_element_located(order_summary_locator))
assert "Order Summary" in order_summary.text

confirm_button_locator = (By.CSS_SELECTOR, "#confirm-button")
confirm_button = WebDriverWait(driver, 10).until(EC.element_to_be_clickable(confirm_button_locator))
confirm_button.click()

# Expected Result: The order is successfully placed, and the user is redirected to the order confirmation page.
order_confirmation_locator = (By.ID, "order-confirmation")
order_confirmation = WebDriverWait(driver, 10).until(EC.visibility_of_element_located(order_confirmation_locator))
assert "Order Confirmation" in order_confirmation.text

# Close the webdriver
driver.quit()
```
